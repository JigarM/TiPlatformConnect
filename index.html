The basic usage

var platform = require(platform).Platform({
  consumerKey: 'XXXXXXXXXX',
  consumerSecret: 'XXXXXXXXXX'.
  accessTokenKey: Ti.App.Properties.getString('platformAccessTokenKey', ''),
  accessTokenSecret: Ti.App.Properties.getString('platformAccessTokenSecret', '')
});

platform.addEventListener('login', function(e){
  if (e.success) {
    Ti.App.Properties.setString('platformAccessTokenKey', e.accessTokenKey);
    Ti.App.Properties.setString('platformAccessTokenSecret', e.accessTokenSecret);

    platform.request('path/to/api', {}, {}, 'GET', function(e){
      if (e.success) {
        var json = JSON.parse(e.result.text);

        // success proc...
      } else {
        // error proc...
      }
  } else {
    // error proc...
  }
});

platform.authorize();

platform.request() params

platform.request('path/to/api', {}, {}, 'GET', function(e){ ... });
  • 1st arg : API path, OK so may write URL full path(http://example.com/path/to/api)
  • 2nd arg : API POST/GET params
  • 3rd arg : customize HTTP header
  • 4th arg : HTTP method
  • 5th arg : callback function

Logout method

twitter.logout(function(){
  Ti.App.Properties.setString('twitterAccessTokenKey', '');
  Ti.App.Properties.setString('twitterAccessTokenSecret', '');
});
